~GET ALL MOVIES IN DATABASE~
SELECT * FROM Movies;

~GET MOVIES WITH A GENRE OF ACTION~
SELECT *
FROM Movies
WHERE Movies.Genre = "Action";

~GET THE MOVIES SORTED ALPHABETICALLY~
SELECT *
FROM Movies
ORDER BY Title;
